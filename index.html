Ingeniero Electrónico con Maestría en Mecatrónica (PUCP) y Candidato a Doctor en Ingeniería Eléctrica por la Universidad Estatal de Campinas (UNICAMP, Brasil). Python · Análisis NDWI con Sentinel-2 
import numpy as np
import rasterio
import matplotlib.pyplot as plt

# Cargar bandas de Sentinel-2
# Banda 3 (Verde, B03) y Banda 8 (NIR, B08)
with rasterio.open("B03.tif") as b3:
    green = b3.read(1).astype(float)

with rasterio.open("B08.tif") as b8:
    nir = b8.read(1).astype(float)

# Calcular NDWI
# NDWI = (Green - NIR) / (Green + NIR)
ndwi = (green - nir) / (green + nir + 1e-10)

# Visualizar
plt.figure(figsize=(10, 6))
plt.imshow(ndwi, cmap='RdYlGn', vmin=-1, vmax=1)
plt.colorbar(label='NDWI')
plt.title('Índice de Agua Normalizado (NDWI)')
plt.axis('off')
plt.tight_layout()
plt.savefig("ndwi_output.png", dpi=150)
plt.show()
Python · CNN para clasificación con Keras 
from tensorflow import keras
from tensorflow.keras import layers

def build_cnn(input_shape=(224, 224, 3), num_classes=2):
    model = keras.Sequential([
        layers.Conv2D(32, (3, 3), activation='relu',
                      input_shape=input_shape),
        layers.MaxPooling2D(2, 2),
        layers.Conv2D(64, (3, 3), activation='relu'),
        layers.MaxPooling2D(2, 2),
        layers.Conv2D(128, (3, 3), activation='relu'),
        layers.GlobalAveragePooling2D(),
        layers.Dense(256, activation='relu'),
        layers.Dropout(0.5),
        layers.Dense(num_classes, activation='softmax')
    ])
    model.compile(
        optimizer='adam',
        loss='sparse_categorical_crossentropy',
        metrics=['accuracy']
    )
    return model

model = build_cnn()
model.summary()
